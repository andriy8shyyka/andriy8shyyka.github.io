];


function renderCatalog() {

    const searchQuery = document.getElementById('search').value.toLowerCase();

    const catalog = document.getElementById('catalog');

    catalog.innerHTML = '';

    catalogData

        .filter(item => item.name.toLowerCase().includes(searchQuery) || item.description.toLowerCase().includes(searchQuery))

        .forEach(item => {

            catalog.innerHTML += `

                <div class="card">

                    <img src="${item.image}" alt="${item.name}">

                    <h3>${item.name}</h3>

                    <p>${item.description}</p>

                </div>

            `;

        });

}


function addItem(event) {

    event.preventDefault();

    const name = document.getElementById('name').value;

    const image = document.getElementById('image').value;

    const description = document.getElementById('description').value;


    catalogData.push({name, image, description});

    localStorage.setItem('fundukCatalog', JSON.stringify(catalogData));

    renderCatalog();


    event.target.reset();

}
